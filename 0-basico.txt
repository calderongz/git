# Comandos basicos de git

# identificar usuario del repositorio de git
git config --global user.email "calderongz@gmail.com"
git config --global user.name "Marco Iñiguez"

# iniciar un repositorio de git
git init

# consultar el estado del repositorio de git
git status

# añadir archivos al el repositorio de git
git add archivo.txt
git add . # añadir todos los archivos del repositorio

# confirmar cambios en el repositorio de git (commit)
git commit -m "Comandos basico de git" 

# consultar commits
git log
